17 bool ConfigParser::load(std::string_view filename) {
18  std::ifstream file{std::string(filename)};
19  if (!file.is_open()) return false;
20  std::string line;
21  while (std::getline(file, line)) {
22  if (line.empty() || line[0] == '#') continue;
23  auto pos = line.find('=');
24  if (pos == std::string::npos) continue;
25  std::string key = line.substr(0, pos);
26  std::string val = line.substr(pos + 1);
27  params_[key] = val;
28  }
29  return true;
30 }
